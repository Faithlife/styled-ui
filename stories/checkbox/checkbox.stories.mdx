# Checkbox

## Variations

<Canvas>
	<Story
		name="Default Theme"
		args={{ isChecked: false, title: 'Click me', type: 'button' }}
		args={resetArgTypes}
	>
		{args => {
			const [_, updateArgs] = useArgs();
			return (
				<Checkbox
					{...args}
					isChecked={args.isChecked === 'mixed' ? args.isChecked : JSON.parse(args.isChecked)}
					onClick={() => updateArgs({ isChecked: !args.isChecked })}
				/>
			);
		}}
	</Story>
</Canvas>
